44  /**
45   * Generates JSW based daemon wrappers.
46   *
47   * @author <a href="mailto:trygvis@inamo.no">Trygve Laugst&oslash;l</a>
48   */
49  @Mojo( name = "generate-daemons", defaultPhase = LifecyclePhase.PACKAGE, requiresDependencyResolution = ResolutionScope.RUNTIME, threadSafe = true )
50  public class GenerateDaemonsMojo
51      extends AbstractScriptGeneratorMojo
52  {
53  
54      // -----------------------------------------------------------------------
55      // Parameters
56      // -----------------------------------------------------------------------
57  
58      /**
59       * The base directory of the project.
60       */
61      @Parameter( defaultValue = "${basedir}", required = true )
62      private File basedir;
63  
64      /**
65       * Set of {@linkplain Daemon}s to generate.
66       */
67      @Parameter( required = true )
68      private Set<Daemon> daemons;
69  
70      /**
71       * {@linkplain JvmSettings} describing min/max memory and stack size, system properties and extra arguments.
72       */
73      @Parameter
74      private JvmSettings defaultJvmSettings;
75  
76      /**
77       * Path (relative to <code>assembleDirectory</code>) of the desired output repository.
78       *
79       * @since 1.5
80       * @todo Synchronize the default value with the other mojos in 2.0
81       */
82      @Parameter( defaultValue = "lib" )
83      private String repositoryName;
84  
85      /**
86       * Target directory for generated daemons.
87       */
88      @Parameter( defaultValue = "${project.build.directory}/generated-resources/appassembler", required = true )
89      private File target;
90  
91      /**
92       * When set to true different platforms
93       * will be output to their own directory (named by platform)
94       * within the target directory.
95       * When set to false, all platforms are output to the same target directory.
96       *
97       * @Since 2.0.0
98       */
99      @Parameter(defaultValue = "true")
100     private boolean separateTargetPlatforms;
101 
102     /**
103      * When enable, name wrapper configuration file as wrapper-${daemon.id}.conf
104      *
105      * @since 1.3
106      */
107     @Parameter( defaultValue = "false" )
108     private boolean useDaemonIdAsWrapperConfName;
109 
110     /**
111      * When enable, prefix the wrapper executable as ${daemon.id}. Otherwise, use the original name( ie wrapper )
112      *
113      * @since 1.8
114      */
115     @Parameter( defaultValue = "false" )
116     private boolean useDaemonIdAsWrapperExePrefixName;
117 
118     /**
119      * Use this option to override the current built-in delta pack binary. You will need to unpack your delta pack
120      * version to a known location set by this option
121      *
122      * @since 1.4.0
123      */
124     @Parameter( property = "externalDeltaPackDirectory" )
125     private File externalDeltaPackDirectory;
126 
127     /**
128      * Use this option to pre insert a content of a known file into the generated wrapper config file. For example:
129      * $include ../conf/another-wrapper.conf
130      *
131      * @since 1.7.0
132      */
133     @Parameter( property = "preWrapperConf" )
134     private File preWrapperConf;
135 
136     /**
137      * Set the name of the generated script name in bin folder instead
138      * of using {@link Daemon#getId()}.
139      *
140      */
141     @Parameter
142     private String binFileName;
143 
144     // -----------------------------------------------------------------------
145     // AbstractMojo Implementation
146     // -----------------------------------------------------------------------
147 
148     /**
149      * calling from Maven.
150      *
151      * @see org.apache.maven.plugin.AbstractMojo#execute()
152      */
153     public void execute()
154         throws MojoExecutionException, MojoFailureException
155     {
156 
157         if ( useWildcardClassPath && !repositoryLayout.equalsIgnoreCase( "flat" ) )
158         {
159             throw new MojoExecutionException( "The useWildcardClassPath works only in"
160                 + " combination with repositoryLayout flat." );
161         }
162 
163         // Set the extensions for bin files for the different platforms
164         setBinFileExtensions();
165 
166         if ( preClean )
167         {
168             removeDirectory( target );
169         }
170 
171         for ( Daemon daemon : daemons )
172         {
173             // -----------------------------------------------------------------------
174             // Load the optional template daemon descriptor
175             // -----------------------------------------------------------------------
176 
177             File descriptor = null;
178 
179             if ( !StringUtils.isEmpty( daemon.getDescriptor() ) )
180             {
181                 descriptor = new File( basedir, daemon.getDescriptor() );
182             }
183 
184             // -----------------------------------------------------------------------
185             //
186             // -----------------------------------------------------------------------
187 
188             org.codehaus.mojo.appassembler.model.JvmSettings modelJvmSettings = null;
189 
190             if ( defaultJvmSettings != null )
191             {
192                 modelJvmSettings = convertJvmSettings( defaultJvmSettings );
193             }
194 
195             ArtifactRepositoryLayout artifactRepositoryLayout = getArtifactRepositoryLayout();
196 
197             // -----------------------------------------------------------------------
198             // Create a daemon object from the POM configuration
199             // -----------------------------------------------------------------------
200 
201             org.codehaus.mojo.appassembler.model.Daemon modelDaemon = convertDaemon( daemon, modelJvmSettings );
202 
203             // -----------------------------------------------------------------------
204             // Default Handling for license file
205             // -----------------------------------------------------------------------
206             if ( this.licenseHeaderFile != null && modelDaemon.getLicenseHeaderFile() == null)
207             {
208                 // Allow overwrite if not set otherwise the set license header file will be used.
209                 modelDaemon.setLicenseHeaderFile( this.licenseHeaderFile.toString() );
210             }
211 
212             modelDaemon.setConfigurationDirectory( configurationDirectory );
213             modelDaemon.setEnvironmentSetupFileName( environmentSetupFileName );
214             modelDaemon.setRepositoryName( repositoryName );
215             modelDaemon.setUseTimestampInSnapshotFileName( useTimestampInSnapshotFileName );
216             modelDaemon.setUseDaemonIdAsWrapperConfName( useDaemonIdAsWrapperConfName );
217             modelDaemon.setUseDaemonIdAsWrapperExePrefixName( useDaemonIdAsWrapperExePrefixName );
218             modelDaemon.setUseWildcardClassPath( useWildcardClassPath );
219 
220             if ( endorsedDir != null )
221             {
222                 modelDaemon.setEndorsedDir( endorsedDir );
223             }
224 
225             if ( preWrapperConf != null )
226             {
227                 modelDaemon.setPreWrapperConf( preWrapperConf.getAbsolutePath() );
228             }
229 
230             if ( this.unixScriptTemplate != null )
231             {
232                 modelDaemon.setUnixScriptTemplate( unixScriptTemplate );
233             }
234             if ( this.windowsScriptTemplate != null )
235             {
236                 modelDaemon.setWindowsScriptTemplate( windowsScriptTemplate );
237             }
238 
239             if ( this.externalDeltaPackDirectory != null )
240             {
241                 modelDaemon.setExternalDeltaPackDirectory( this.externalDeltaPackDirectory.getAbsolutePath() );
242             }
243 
244             // -----------------------------------------------------------------------
245             //
246             // -----------------------------------------------------------------------
247 
248             for ( String platform : daemon.getPlatforms() )
249             {
250                 final File output;
251                 if  ( separateTargetPlatforms ) {
252                     output = new File(target, platform);
253                 } else {
254                     output = target;
255                 }
256 
257                 DaemonGenerationRequest request = new DaemonGenerationRequest();
258 
259                 // TODO: split platform from generator (platform = operating systems, generator = jsw, booter,
260                 // standard). Generator is a property of the daemon itself
261                 request.setPlatform( platform );
262                 request.setStubDescriptor( descriptor );
263                 request.setStubDaemon( modelDaemon );
264                 request.setOutputDirectory( output );
265                 request.setMavenProject( mavenProject );
266                 request.setLocalRepository( localRepository );
267                 request.setRepositoryLayout( artifactRepositoryLayout );
268                 request.setOutputFileNameMapping( this.outputFileNameMapping );
269                 request.setBinScriptName( binFileName );
270 
271                 try
272                 {
273                     daemonGeneratorService.generateDaemon( request );
274                 }
275                 catch ( DaemonGeneratorException e )
276                 {
277                     throw new MojoExecutionException( "Error while generating daemon.", e );
278                 }
279 
280                 File outputDirectory = new File( request.getOutputDirectory(), daemon.getId() );
281 
282                 // ----------------------------------------------------------------------
283                 // Install dependencies in the new repository
284                 // ----------------------------------------------------------------------
285                 super.installDependencies( outputDirectory.getAbsolutePath(), repositoryName );
286 
287                 // ----------------------------------------------------------------------
288                 // Copy configuration directory
289                 // ----------------------------------------------------------------------
290 
291                 if ( this.preAssembleDirectory != null && this.preAssembleDirectory.isDirectory() )
292                 {
293                     doCopyPreAssembleDirectory( outputDirectory.getAbsolutePath() );
294                 }
295 
296                 if ( this.copyConfigurationDirectory && configurationSourceDirectory.isDirectory() )
297                 {
298                     doCopyConfigurationDirectory( outputDirectory.getAbsolutePath() );
299                 }
300 
301                 // ----------------------------------------------------------------------
302                 // Create logs and temp dirs if specified
303                 // ----------------------------------------------------------------------
304 
305                 doCreateExtraDirectories( outputDirectory );
306             }
307 
308         }
309     }

311     // TODO: see if it is possible to just inherit from the model daemon
312     private org.codehaus.mojo.appassembler.model.Daemon convertDaemon( Daemon daemon,
313                                                                        org.codehaus.mojo.appassembler.model.JvmSettings modelJvmSettings )
314     {
315         org.codehaus.mojo.appassembler.model.Daemon modelDaemon;
316 
317         modelDaemon = new org.codehaus.mojo.appassembler.model.Daemon();
318 
319         modelDaemon.setId( daemon.getId() );
320         modelDaemon.setMainClass( daemon.getMainClass() );
321         modelDaemon.setWrapperMainClass( daemon.getWrapperMainClass() );
322         modelDaemon.setWrapperLogFile( daemon.getWrapperLogFile() );
323         modelDaemon.setCommandLineArguments( daemon.getCommandLineArguments() );
324         modelDaemon.setConfigurationDirectory( daemon.getConfigurationDirectory() );
325         modelDaemon.setLicenseHeaderFile( daemon.getLicenseHeaderFile() );
326         modelDaemon.setShowConsoleWindow( daemon.isShowConsoleWindow() );
327         modelDaemon.setEnvironmentSetupFileName( daemon.getEnvironmentSetupFileName() );
328         modelDaemon.setRepositoryName( daemon.getRepositoryName() );
329         modelDaemon.setEndorsedDir( daemon.getEndorsedDir() );
330         modelDaemon.setPreWrapperConf( daemon.getPreWrapperConf() );
331         modelDaemon.setName( daemon.getName() );
332         modelDaemon.setLongName( daemon.getLongName() );
333 
334         if ( daemon.getJvmSettings() != null )
335         {
336             modelDaemon.setJvmSettings( convertJvmSettings( daemon.getJvmSettings() ) );
337         }
338         else
339         {
340             modelDaemon.setJvmSettings( modelJvmSettings );
341         }
342 
343         if ( daemon.getGeneratorConfigurations() != null )
344         {
345             modelDaemon.setGeneratorConfigurations( convertGeneratorConfigurations( daemon.getGeneratorConfigurations() ) );
346         }
347 
348         return modelDaemon;
349     }
350 
351     private List<org.codehaus.mojo.appassembler.model.GeneratorConfiguration> convertGeneratorConfigurations(List<GeneratorConfiguration> generatorConfigurations )
352     {
353         List<org.codehaus.mojo.appassembler.model.GeneratorConfiguration> value =
354             new ArrayList<org.codehaus.mojo.appassembler.model.GeneratorConfiguration>( generatorConfigurations.size() );
355         for ( GeneratorConfiguration config : generatorConfigurations )
356         {
357             value.add( convertGeneratorConfiguration( config ) );
358         }
359         return value;
360     }
361 
362     private org.codehaus.mojo.appassembler.model.GeneratorConfiguration convertGeneratorConfiguration( GeneratorConfiguration config )
363     {
364         org.codehaus.mojo.appassembler.model.GeneratorConfiguration value =
365             new org.codehaus.mojo.appassembler.model.GeneratorConfiguration();
366         value.setGenerator( config.getGenerator() );
367         value.setConfiguration( config.getConfiguration() );
368         value.setIncludes( config.getIncludes() );
369 
370         return value;
371     }
372 
373     // TODO: see if it is possible to just inherit from the model JVM Settings
374     private org.codehaus.mojo.appassembler.model.JvmSettings convertJvmSettings( JvmSettings jvmSettings )
375     {
376         org.codehaus.mojo.appassembler.model.JvmSettings modelJvmSettings =
377             new org.codehaus.mojo.appassembler.model.JvmSettings();
378 
379         modelJvmSettings.setInitialMemorySize( jvmSettings.getInitialMemorySize() );
380         modelJvmSettings.setMaxMemorySize( jvmSettings.getMaxMemorySize() );
381         modelJvmSettings.setMaxStackSize( jvmSettings.getMaxStackSize() );
382         if ( jvmSettings.getSystemProperties() == null )
383         {
384             modelJvmSettings.setSystemProperties( new ArrayList<String>() );
385         }
386         else
387         {
388             modelJvmSettings.setSystemProperties( Arrays.asList( jvmSettings.getSystemProperties() ) );
389         }
390         if ( jvmSettings.getExtraArguments() == null )
391         {
392             modelJvmSettings.setExtraArguments( new ArrayList<String>() );
393         }
394         else
395         {
396             modelJvmSettings.setExtraArguments( Arrays.asList( jvmSettings.getExtraArguments() ) );
397         }
398 
399         return modelJvmSettings;
400     }
