46  /**
47   * @author Andrew Williams
48   * @plexus.component role-hint="daemontools"
49   */
50  public class DaemonToolsDaemonGenerator
51      extends AbstractLogEnabled
52      implements DaemonGenerator
53  {
54      // -----------------------------------------------------------------------
55      // DaemonGenerator Implementation
56      // -----------------------------------------------------------------------
57  
58      /*
59       * (non-Javadoc)
60       * @see org.codehaus.mojo.appassembler.daemon.DaemonGenerator#generate(org.codehaus.mojo.appassembler.daemon.
61       * DaemonGenerationRequest)
62       */
63      public void generate( DaemonGenerationRequest request )
64          throws DaemonGeneratorException
65      {
66          Daemon daemon = request.getDaemon();
67  
68          try
69          {
70              FileUtils.forceMkdir( request.getOutputDirectory() );
71          }
72          catch ( IOException e )
73          {
74              throw new DaemonGeneratorException( "Error creating output directory: " + request.getOutputDirectory(), e );
75          }
76  
77          File envDir = new File( request.getOutputDirectory(), "env" );
78          envDir.mkdir();
79  
80          copyEnvFile( "JAVA_HOME", envDir );
81          copyEnvFile( "USER", envDir );
82  
83          File logDir = new File( request.getOutputDirectory(), "logs" );
84          logDir.mkdir();
85  
86          File serviceDir = new File( request.getOutputDirectory(), "service" );
87          serviceDir.mkdir();
88  
89          // -----------------------------------------------------------------------
90          //
91          // -----------------------------------------------------------------------
92  
93          InputStream in = this.getClass().getResourceAsStream( "run.sh.template" );
94  
95          if ( in == null )
96          {
97              throw new DaemonGeneratorException( "Could not load template." );
98          }
99  
100         InputStreamReader reader = new InputStreamReader( in );
101 
102         Map<Object, Object> context = new HashMap<Object, Object>();
103         context.put( "MAINCLASS", daemon.getMainClass() );
104         context.put( "NAME", daemon.getId() );
105 
106         InterpolationFilterReader interpolationFilterReader = new InterpolationFilterReader( reader, context, "@", "@" );
107 
108         File runFile = new File( request.getOutputDirectory(), "run" );
109         FileWriter out = null;
110 
111         try
112         {
113             // -----------------------------------------------------------------------
114             // Write the file
115             // -----------------------------------------------------------------------
116 
117             out = new FileWriter( runFile );
118 
119             IOUtil.copy( interpolationFilterReader, out );
120         }
121         catch ( IOException e )
122         {
123             throw new DaemonGeneratorException( "Error writing output file: " + runFile.getAbsolutePath(), e );
124         }
125         finally
126         {
127             IOUtil.close( interpolationFilterReader );
128             IOUtil.close( out );
129         }
130 
131     }
132 
133     private void copyEnvFile( String envName, File envDir )
134         throws DaemonGeneratorException
135     {
136         Writer out = null;
137         Reader envReader = null;
138 
139         File envFile = new File( envDir, envName );
140 
141         try
142         {
143             envReader = new InputStreamReader( this.getClass().getResourceAsStream( "env/" + envName ), "UTF-8");
144 
145             // -----------------------------------------------------------------------
146             // Write the file
147             // -----------------------------------------------------------------------
148 
149             out = new FileWriter( envFile );
150 
151             IOUtil.copy( envReader, out );
152         }
153         catch ( IOException e )
154         {
155             throw new DaemonGeneratorException( "Error writing environment file: " + envFile, e );
156         }
157         finally
158         {
159             IOUtil.close( envReader );
160             IOUtil.close( out );
161         }
162     }
163 }
