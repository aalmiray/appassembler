21  package org.codehaus.mojo.appassembler.daemon.generic;
22  
23  import java.io.File;
24  
25  import javax.xml.parsers.DocumentBuilder;
26  import javax.xml.parsers.DocumentBuilderFactory;
27  
28  import org.codehaus.mojo.appassembler.daemon.AbstractDaemonGeneratorTest;
29  import org.w3c.dom.Document;
30  
31  /**
32   * @author <a href="mailto:trygvis@inamo.no">Trygve Laugst&oslash;l</a>
33   */
34  public class GenericDaemonGeneratorTest
35      extends AbstractDaemonGeneratorTest
36  {
37      public void testGenerationWithAllInfoInDescriptor()
38          throws Exception
39      {
40          runTest( "generic", "src/test/resources/project-1/pom.xml", "src/test/resources/project-1/descriptor.xml",
41                   "target/output-1-generic" );
42  
43          File actualAppXml = new File( getTestFile( "target/output-1-generic" ), "app.xml" );
44  
45          assertTrue( "config file is missing: " + actualAppXml.getAbsolutePath(), actualAppXml.isFile() );
46  
47          DocumentBuilderFactory builderFactory = DocumentBuilderFactory.newInstance();
48  
49          builderFactory.setIgnoringComments( true );
50          builderFactory.setIgnoringElementContentWhitespace( true );
51          DocumentBuilder builder = builderFactory.newDocumentBuilder();
52  
53          Document actual = builder.parse( actualAppXml );
54  
55          assertNodeEquals( "com.westerngeco.example.App", "mainClass", actual );
56          assertNodeEquals( "org/codehaus/mojo/appassembler/project-1/1.0-SNAPSHOT/project-1-1.0-SNAPSHOT.jar",
57                            "relativePath", actual );
58          assertNodeEquals( "345", "initialMemorySize", actual );
59          assertNodeEquals( "234", "maxMemorySize", actual );
60          assertNodeEquals( "321", "maxStackSize", actual );
61          assertNodeEquals( "foo=bar", "systemProperty", actual );
62          assertNodeEquals( "arg1=arg1-value", "commandLineArgument", actual );
63          assertNodeNull( "commandLineArgument", actual );
64  
65      }
66  
67      // MAPPASM-143. The old unit tests testGenerationWithSnapshotDependencies() has been converted into
68      // a integration tests to get the integration test running after a release change etc.
69      private void assertNodeEquals( String expected, String tagName, Document document )
70      {
71          assertEquals( "Node with tag name " + tagName + " does not match", expected,
72                        document.getElementsByTagName( tagName ).item( 0 ).getFirstChild().getNodeValue() );
73      }
74  
75      private void assertNodeNull( String tagName, Document document )
76      {
77          assertNull( document.getElementsByTagName( tagName ).item( 1 ).getFirstChild() );
78      }
79  }
